--funciones de oracle
DECLARE
   FECHA DATE; --almacena fecha
   FECHA_HORA TIMESTAMP; --almacena fecha y hora
BEGIN
   FECHA:=SYSDATE;
   DBMS_OUTPUT.put_line('LA FECHA OBTENIDA CON SYSDATE ES: '||FECHA);
   FECHA_HORA:=SYSTIMESTAMP;
   DBMS_OUTPUT.put_line('LA FECHA Y HORA OBTENIDA CON SYSDATE ES: '||FECHA_HORA);
   --recibe como parametro la fecha y formato
   FECHA:= TO_DATE('2020-12-23','YYYY-MM-DD');
   DBMS_OUTPUT.put_line('LA FECHA ASIGNADA CON TO_DATE ES: '||FECHA);
   
   FECHA_HORA:= TO_TIMESTAMP('25-05-1999 23:30:16','DD-MM-YYYY HH24:MI:SS');
   DBMS_OUTPUT.put_line('LA FECHA Y HORA ASIGNADA CON TO_TIMESTAMP ES: '||FECHA_HORA);   
   --cambiar el formato en el que se muestra la fecha
   DBMS_OUTPUT.put_line('LA FECHA USANDO TO_CHAR ES: '||TO_CHAR(FECHA,'DD-MM-YYYY'));
   
   DBMS_OUTPUT.put_line('LA FECHA Y HORA USANDO TO_CHAR ES: '||TO_CHAR(FECHA_HORA,'DD-MM-YYYY HH-MI-SS AM'));
   
   
   --funcion USER
   DBMS_OUTPUT.PUT_LINE('EL USUARIO CON EL QUE ESTAMOS CONECTADOS ES: '|| USER);
   
END;



--obtener valores de fechas con consultas
SELECT ORDER_DATE, TO_DATE(ORDER_DATE,'YYYYMMDD'),TO_TIMESTAMP(ORDER_DATE,'YYYYMMDD HH:MI:SS') FROM ORDERS;

SELECT ORDER_DATE, TO_CHAR(TO_DATE(ORDER_DATE,'YYYYMMDD'),'YYYY-MM-DD'),
TO_TIMESTAMP(ORDER_DATE,'YYYYMMDD HH:MI:SS') FROM ORDERS;
-- SUBSTR sub string muestra solo ciertos caracteres
