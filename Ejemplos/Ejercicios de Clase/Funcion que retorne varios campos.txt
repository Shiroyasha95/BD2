-- CREAR UNA FUNCION QUE OBTENGA LOS CAMPOS NOMBRE DE TIENDA, TELEFONO, CORREO Y CALLE DE LA TABLA STORES EN BASE AL ID DE TIENDA
CREATE OR REPLACE FUNCTION FN_DATOS_TIENDA(ID_TIENDA STORES.STORE_ID%TYPE, TEL OUT STORES.PHONE%TYPE, CORREO OUT STORES.EMAIL%TYPE,
CALLE OUT STORES.STREET%TYPE)
RETURN VARCHAR2

IS 
  NOMBRE_TIENDA STORES.STORE_NAME%TYPE;
BEGIN
    SELECT STORE_NAME,PHONE,EMAIL,STREET INTO NOMBRE_TIENDA,TEL,CORREO,CALLE FROM STORES WHERE STORE_ID=ID_TIENDA;
    
    RETURN NOMBRE_TIENDA;
END;

DECLARE
     NOMBRE_TIENDA STORES.STORE_NAME%TYPE;
     TEL STORES.PHONE%TYPE;
     CORREO STORES.EMAIL%TYPE;
     CALLE STORES.STREET%TYPE;
BEGIN
   NOMBRE_TIENDA:=FN_DATOS_TIENDA(2,TEL,CORREO,CALLE);
   DBMS_OUTPUT.put_line('EL NOMBRE DE LA TIENDA ES: '||NOMBRE_TIENDA);
   DBMS_OUTPUT.put_line('EL TELEFONO DE LA TIENDA ES: '||TEL);
   DBMS_OUTPUT.put_line('EL CORREO DE LA TIENDA ES: '||CORREO);
   DBMS_OUTPUT.put_line('LA DIRECCION DE LA TIENDA ES: '||CALLE);
END;
