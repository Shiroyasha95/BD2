--cursores ejemplo 1
--se utiliza cuando queremos recorrer varios registros de una consulta puede ser implicito o explicito
DECLARE
   NOMBRE_PRODUCTO PRODUCTS.PRODUCT_NAME%TYPE;
   CURSOR CNOMBRE_PRODUCTO IS SELECT PRODUCT_NAME FROM PRODUCTS;
 -- ITERACION NUMBER(10);
BEGIN
--  SELECT PRODUCT_NAME INTO NOMBRE FROM PRODUCTS WHERE PRODUCT_ID=1;
--  DBMS_OUTPUT.put_line('EL NOMBRE DEL PRODUCTO CON ID 1 ES '||NOMBRE);
-- DBMS_OUTPUT.PUT_LINE('LA CANTIDAD DE REGOSTROS ES: '||SQL%ROWCOUNT);--cursor implicito
   OPEN CNOMBRE_PRODUCTO;
   LOOP
   --FETCH posiciona los registros 
   FETCH CNOMBRE_PRODUCTO INTO NOMBRE_PRODUCTO;
   --NOTFOUND retorna true o false, se deteiene cuando ya no hay mas registros 
   EXIT WHEN CNOMBRE_PRODUCTO%NOTFOUND;
   DBMS_OUTPUT.PUT_LINE('EL NOMBRE DEL PRODUCTO ES: '|| NOMBRE_PRODUCTO);
   DBMS_OUTPUT.PUT_LINE('LA CANTIDAD DE REGISTROS ES: '||CNOMBRE_PRODUCTO%ROWCOUNT);
   END LOOP;
   CLOSE CNOMBRE_PRODUCTO;
END;
