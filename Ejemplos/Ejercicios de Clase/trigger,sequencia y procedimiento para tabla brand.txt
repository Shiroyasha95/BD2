--CREAR UN PROCEDIMIENTO PARA TRABAJAR CON ELSE

CREATE SEQUENCE SQ_BRAND
START WITH 16
INCREMENT BY 1;

CREATE OR REPLACE TRIGGER TG_SQ_BRANDS
BEFORE INSERT ON BRANDS
FOR EACH ROW 
DECLARE
BEGIN
  :NEW.BRAND_ID:=SQ_BRAND.NEXTVAL;
END;


CREATE OR REPLACE PROCEDURE PS_INSERTA_BRANDS(BRAND_NOMBRE VARCHAR2, ID_BRAND OUT NUMBER)
IS
BEGIN 
   INSERT INTO BRANDS(BRAND_NAME) VALUES(BRAND_NOMBRE);
   SELECT BRAND_ID INTO ID_BRAND FROM BRANDS WHERE BRAND_NAME=BRAND_NOMBRE;
   COMMIT;
END;
