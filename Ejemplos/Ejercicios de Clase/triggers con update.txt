CREATE OR REPLACE TRIGGER TG_INSERTA_BRANDS
BEFORE INSERT ON BRANDS
FOR EACH ROW
DECLARE
--TRIM() eliminar los espacios de las cadenas
--UPPER() pasar a mayusculas
BEGIN 
 :NEW.BRAND_NAME:=TRIM(UPPER(:NEW.BRAND_NAME));
END;
