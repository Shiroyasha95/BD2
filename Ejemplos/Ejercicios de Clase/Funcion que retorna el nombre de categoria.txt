--FUNCION QUE ME RETORNE EL NOMBRE DE LA CATEGORIA

CREATE OR REPLACE FUNCTION FN_NOMBRE_CAT (ID_CAT NUMBER, NOMBRE_CATE OUT VARCHAR2)
RETURN VARCHAR2
IS
   NOMBRE CATEGORIES.CATEGORY_NAME%TYPE;
   
BEGIN

   SELECT CATEGORIES.CATEGORY_NAME INTO NOMBRE FROM CATEGORIES WHERE CATEGORY_ID = ID_CAT;
   NOMBRE_CATE:=NOMBRE;
RETURN NOMBRE;
END;

DECLARE 
   NOMBRE_FUNCION CATEGORIES.CATEGORY_NAME%TYPE;
   NOMBRE_PARAMETRO CATEGORIES.CATEGORY_NAME%TYPE;
BEGIN 
  NOMBRE_FUNCION:= FN_NOMBRE_CAT(4,NOMBRE_PARAMETRO);
  DBMS_OUTPUT.put_line('EL NOMBRE DE LA CATEGORIA QUE RETORNA LA FUNCION ES: '|| NOMBRE_FUNCION);
  DBMS_OUTPUT.put_line('EL NOMBRE DE LA CATEGORIA ES QUE RETORNA EL PARAMETRO ES : '|| NOMBRE_PARAMETRO);
END;
