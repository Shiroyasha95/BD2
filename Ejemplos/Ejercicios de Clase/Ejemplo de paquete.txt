-- CREAR PAQUETE que obtenga info de base de datos
CREATE OR REPLACE PACKAGE PK_INF_VENTA_BICICLETAS
IS
    NOMBRE VARCHAR2(500);
    --prototipo de la funcion
    FUNCTION FN_DATOS_TIENDA(ID_TIENDA STORES.STORE_ID%TYPE) RETURN VARCHAR2; 
    --prototipo de un procedimiento
    PROCEDURE SP_INSERTA_BRANDS (NOMBRE_MARCA BRANDS.BRAND_NAME%TYPE);
    
END;

-- CREAR EL CUERPO DEL PAQUETE 
CREATE OR REPLACE PACKAGE BODY PK_INF_VENTA_BICICLETAS
IS
    FUNCTION FN_DATOS_TIENDA(ID_TIENDA STORES.STORE_ID%TYPE) RETURN VARCHAR2
    IS
    NOMBRE_TIENDA STORES.STORE_NAME%TYPE;
    BEGIN
       SELECT STORE_NAME INTO NOMBRE_TIENDA FROM STORES  WHERE  STORE_ID =  ID_TIENDA;
       NOMBRE:=NOMBRE_TIENDA;
       RETURN NOMBRE;
    END;
    
    
    PROCEDURE SP_INSERTA_BRANDS (NOMBRE_MARCA BRANDS.BRAND_NAME%TYPE)
    IS
    
    BEGIN
        INSERT INTO BRANDS (BRAND_NAME)VALUES(NOMBRE_MARCA);
        COMMIT;
    END;
END;



DECLARE
    NOMBRE VARCHAR2(500);
BEGIN 
   NOMBRE:= PK_INF_VENTA_BICICLETAS.FN_DATOS_TIENDA(2);
   DBMS_OUTPUT.put_line('EL NOMBRE DE LA TIENDA ES: '||NOMBRE);
   PK_INF_VENTA_BICICLETAS.SP_INSERTA_BRANDS('DIUNSA');
   
END;


SET SERVEROUTPUT ON;
